NYT's Wordle. 

to run this game cd into the wordle directory and then type:
"python main.py"

# data types:

Word:
    text        # the word as a string
    letter_list # list of letters in the word

    constructor(t):
        text = t
        split_word(text)

    split_word(text):
        letter_list = empty list
        for each letter c in text:
            append c to letter_list

    contains_letter(letter)
        for each c in letter_list:
            if c == letter:
                return true
        return false

    index_match(letter, index)
        if letter == letter_list[index]:
            return true
        otherwise: 
            return false

    get_letter_list():

    get_text():
        return text
